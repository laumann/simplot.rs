This macro emulates an "any-arity" free function that zips iterators

Examples

#[macro_use]
extern crate zip;

let chars = ['a', 'b', 'c'];
let mut v = vec![0, 1, 2];

for (&c, i, &mut j) in zip!(&chars, 0..5, &mut v) {
    assert!(i < 3);
    assert_eq!(i, j);
}
